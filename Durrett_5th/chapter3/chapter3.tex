\section{The De Moivre-Laplace Theorem}

\begin{exercise}
    Generalize the proof of Lemma 3.1.1 to conclude that if $\max_{1\le j\le n} \abs{c_{j,n}} \to 0$,
    $\sum_{j=1}^n c_{j,n}\to \lambda$ and $\sup_n \sum_{j=1}^{n} \abs{c_{j,n}}<\infty$ then $\prod_{j=1}^n (1+c_{j,n})\to e^\lambda$.
\end{exercise}
\begin{solution}
    By the Taylor's expansion, the logritham of the sum is
    \begin{align*}
        \sum_{j=1}^{n} \ln(1+c_{j,n}) & = \sum_{j=1}^{n} (c_{j,n} - \frac{1}{(1+c_{j,n})^2}c_{j,n}^2)             \\
                                      & =\sum_{j=1}^{n} c_{j,n} - \sum_{j=1}^{n} \frac{1}{(1+c_{j,n})^2}c_{j,n}^2
    \end{align*}
    where $\sum_{j=1}^{n} c_{j,n} \to \lambda$.And letting $c_n := \max_{1\le j\le n} \abs{c_{j,n}}<$, for large $n$ such that $c_n<1$
    \[
        \abs{\sum_{j=1}^{n} \frac{1}{(1+c_{j,n})^2}c_{j,n}^2} \le \abs{\frac{1}{(1-c_n)^2}}\abs{c_n}\sum_{j=1}^{n}{\abs{c_{j,n}}} \to 0
    \]
    The desired result follows.
\end{solution}
